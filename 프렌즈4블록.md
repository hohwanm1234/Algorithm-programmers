package org.example;

/**
 * <pre>
 *     프렌즈4블록
 * </pre>
 *
 * @author 문호환
 * @since 2024-04-29
 */

public class Main {
    public static void main(String[] arg) {
        Solution s = new Solution();

        // TestCase1
        int m = 4;
        int n = 5;
        String[] board = {"CCBDE", "AAADE", "AAABF", "CCBBF"};
        System.out.println(s.solution(m, n, board));

        s.resetAnswer();

        int m2 = 6;
        int n2 = 6;
        String[] board2 = {"TTTANT", "RRFACC", "RRRFCC", "TRRRAA", "TTMMMF", "TMMTTJ"};
        System.out.println(s.solution(m2, n2, board2));
    }

    public static class Solution {
        int answer = 0;

        public void resetAnswer() {
            answer = 0;
        }

        int solution(int m, int n, String[] board) {
            answer = 0;
            char[][] charMap = new char[m][n];

            for (int i = 0; i < m; i++) {
                charMap[i] = board[i].toCharArray();
            }

            while (true) {
                int count = check(charMap, m, n);
                if (count == 0) {
                    break;
                }

                delete(charMap, m, n);
                answer += count;
            }

            return answer;
        }

        /**
         * 삭제 블록 확인
         * @return count (삭제 블록 갯수)
         * 빈블럭 = 'A'로 처리
         */
        private int check(char[][] charMap, int m, int n) {
            int count = 0;
            boolean[][] isChecked = new boolean[m][n]; // 체크 여부 확인

            for (int i = 0; i < m - 1; i++) {
                for (int j = 0; j < n - 1; j++) {
                    if (charMap[i][j] == 'A') { // 블록이 존재하지 않는 경우
                        continue;
                    }

                    char checkMap = charMap[i][j];

                    for (int k = i; k < i + 2; k++) {
                        for (int l = j; l < j + 2; l++) {
                            if (charMap[i][j] != checkMap)
                                break;
                            isChecked[k][l] = true;
                        }
                    }
                }
            }

            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    if (isChecked[i][j]) {
                        charMap[i][j] = 'A';
                        count++;
                    }
                }
            }

            return count;
        }

        /**
         * 블록 삭제 처리
         */
        private void delete(char[][] charMap, int m, int n) {
            for (int i = 0; i < n; i++) {
                for (int j = m - 1; j >= 0; j--) {
                    if (charMap[j][i] == 'A') {
                        for (int k = j - 1; k >= 0; k--) {
                            if (charMap[k][i] != 'A') {
                                charMap[j][i] = charMap[k][i];
                                charMap[k][i] = 'A';
                                break;
                            }
                        }
                    }
                }
            }
        }
    }
}
